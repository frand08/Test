187  SERIAL_SEND("FLASH init error. Resetting MCU\r\n");
188  /* TODO: Reset MCU or something... */
189  osThreadTerminate(UARTTaskHandle);
190  }

200  /* Wait until UART data received */
201  event = osMessageGet(UARTQueueHandle, osWaitForever);
202  if(event.status == osEventMessage)

255 void StartMagTask(void const * argument)
256 {
257  LIS3MDL_DataTypeDef read_data;
258  LIS3MDL_StatusTypeDef magnetometer_retval = LIS3MDL_ERROR;
259  uint32_t memory_id = 0;
260 
261  /* Infinite loop */
262  for(;;)
263  {
264  /* Take SPI semaphore when available */
265  if(osSemaphoreWait(SPISemaphoreHandle, osWaitForever) > 0)
266  {
267  /* Read magnetometer values */
268  magnetometer_retval = LIS3MDL_ReadValues(&read_data);
269 
270  /* Release SPI semaphore */
271  osSemaphoreRelease(SPISemaphoreHandle);
272  }
273 
274  /* In case a UART receive interrupt occured during the LIS3MDL read, release the SPI
275  * semaphore to unlock the serial thread. An osDelay is added to call the scheduler */
276  osDelay(1);
277 
278  /* Try to write only if magnetometer data read OK */
279  if(magnetometer_retval == LIS3MDL_OK)
280  {
281  /* Take SPI semaphore when available */
282  if(osSemaphoreWait(SPISemaphoreHandle, osWaitForever) > 0)
283  {
284  /* Write to FLASH memory and if OK increment offset */
285  if(EXTFLASH_WriteData(memory_id, read_data.mag_x, read_data.mag_y, read_data.mag_z, read_data.temp) == EXTFLASH_OK)
286  {
287  /* Increment memory offset */
288  memory_id++;
289  }
290 
291  /* Release SPI semaphore */
292  osSemaphoreRelease(SPISemaphoreHandle);
293  }
294  }
295  /* Send data every second (if possible) */
296  osDelay(1000);
297  }
